\section{Комбинаторы парсеров}

\begin{frame}[fragile]
  \frametitle{Парсер}
  \begin{semiverbatim}
    trait Parser[+T]\pause{} extends Input => ParserResult[T]
  \end{semiverbatim}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Парсер}
  \begin{semiverbatim}
    trait Parser[+T] extends {\color{red}String} => ParserResult[T]
  \end{semiverbatim}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Результат}
  \begin{semiverbatim}
    trait ParseResult[+T]
    \pause
    case class Success[+T](value: T, input: String)
      extends ParseResult[T]
    \pause
    case class Failure[+T](message: String, input: String) 
      extends ParseResult[T]
  \end{semiverbatim}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Пример использования}
  \begin{semiverbatim}
    \uncover<1->{def IntParser: Parser[Int]
    \uncover<2->{IntParser("10")\uncover<3->{ match \{
      \uncover<4->{case Success(value, _) => handle(value)}
      \uncover<5->{case Failure(message, _) => println(message)}
    \}}}}
  \end{semiverbatim}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Комбинатор парсеров}
  \begin{semiverbatim}
    \uncover<1->{def combine\uncover<2->{[T](\uncover<3->{p1: Parser[T],} \uncover<4->{p2: Parser[T]})}\uncover<5->{: Parser[T]}}
  \end{semiverbatim}
\end{frame}
